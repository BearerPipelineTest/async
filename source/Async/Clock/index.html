def self.measure

	yield
	
	return self.now - start_time

def stop!

Stop measuring a duration and append the duration to the current total.

Implementation

 
def stop!
	if @started
		@total += (Clock.now - @started)
		@started = nil
	end
	
	return @total
end

def total

The total elapsed time including any current duration.

Implementation

 
def total
	total = @total
	
	if @started
		total += (Clock.now - @started)
	end
	
	return total
end
