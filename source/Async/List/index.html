def insert(item)
	unless @tail
		@tail = item
		@head = item
		
		# Consistency:
		item.head = nil
		item.tail = nil
	else
		@head.tail = item
		item.head = @head
		
		# Consistency:
		item.tail = nil
		
		@head = item
	end
	
	@size += 1
	
	return self
end
